import os import re import numpy as np import matplotlib.pyplot as plt try: plt.rcParams["font.sans-serif"] = ["Microsoft JhengHei", "SimHei", "Arial Unicode MS"] plt.rcParams["axes.unicode_minus"] = False except Exception: pass # ⭐ 資料夾 BASE = r"C:\Users\56-000M000-10\ery\data\our_ppg\7-1_V2\test" BASE = r"C:\Users\56-000M000-10\ery\data\our_ppg\7-2-V1" # BASE = r"C:\Users\56-000M000-10\ery\data\our_ppg\7-2" # ── 常數：完全對應 APP 現在的 calculateHb() ── FS = 256 ST = 8 BUFFER_SIZE = ST * FS EPS = 1e-6 EPS_IRO = 1214.0 EPS_IR = 693.44 D_FACTOR = 1.05 TF = 0.00000154 # ⭐ 跟 APP 裡 static constexpr float TF 保持一致，改了記得同步這裡 # ⭐ 是否輸出「處理後、算 Hb 前」的波型圖 PLOT_WAVEFORM = True WAVEFORM_DIR_NAME = "waveforms" # 每筆檔案各自一張圖，存在 BASE/waveforms/ 底下 PLOT_OVERVIEW_GRID = True # 額外輸出一張「全部檔案疊在一起看」的總覽圖 # ============================================================ # 前處理：remove_dc + 線性 detrend，逐行對應 C++ remove_dc() + rf_linear_regression_beta() # ============================================================ def remove_dc(sig: np.ndarray): mean = np.mean(sig) return sig - mean, mean def linear_detrend(sig: np.ndarray): n = len(sig) x = np.arange(n) - (n - 1) / 2.0 beta = np.dot(x, sig) / np.dot(x, x) return sig - beta * x def rf_rms(sig: np.ndarray): return np.sqrt(np.mean(sig ** 2)) def get_hb_features(ir_raw: np.ndarray, red_raw: np.ndarray): """完全對應 APP calculateHb()：remove_dc -> 線性 detrend -> RMS -> A_ir/A_red/R1 多回傳處理後的 ir_ac / red_ac，方便畫「算 Hb 前」的波型。""" ir_raw = np.asarray(ir_raw, dtype=np.float64) red_raw = np.asarray(red_raw, dtype=np.float64) ir_ac, ir_mean = remove_dc(ir_raw) red_ac, red_mean = remove_dc(red_raw) ir_ac = linear_detrend(ir_ac) red_ac = linear_detrend(red_ac) ac_ir = rf_rms(ir_ac) ac_red = rf_rms(red_ac) if abs(ir_mean) < EPS or abs(red_mean) < EPS: return None, ir_ac, red_ac A_ir = ac_ir / abs(ir_mean) A_red = ac_red / abs(red_mean) R1 = A_red / A_ir return (A_ir, A_red, R1), ir_ac, red_ac def calc_hb(A_ir: float, R1: float): """完全對應 APP calculateHb() 的 SpO2 + 單 Tf Beer-Lambert 公式""" R1_adj = R1 - 0.01 spo2 = (6.2357 * R1_adj - 36.1546) * R1_adj + 111.8910 if spo2 < 95.0 or spo2 > 100.0: return None, spo2, "SpO2 超出範圍" spo2_frac = min(max(spo2 / 100.0, 0.0), 1.0) denom = spo2_frac * (EPS_IRO + (1.0 - spo2_frac) * EPS_IR) Chbo = A_ir / (denom * TF) Chb = Chbo * (1.0 - spo2_frac) / spo2_frac hb = 10.0 * D_FACTOR * (Chbo + Chb) if hb < 5.0 or hb > 20.0: return None, spo2, f"Hb 超出生理合理範圍({hb:.2f})" return hb, spo2, None def parse_time_label(fname: str): """從檔名 ppg_20260701_101641_154.npy 取出 10:16:41 這種時間字串，抓不到就回傳原檔名""" m = re.search(r"_(\d{8})_(\d{6})_(\d+)\.npy$", fname) if m: hhmmss = m.group(2) return f"{hhmmss[0:2]}:{hhmmss[2:4]}:{hhmmss[4:6]}" return fname.replace(".npy", "") def plot_processed_waveform(fname, label, ir_ac, red_ac, out_dir, note=""): """畫出『remove_dc + detrend 之後、算 Hb 之前』的 IR / Red 波型（單一檔案）""" t = np.arange(len(ir_ac)) / FS # 秒 fig, axes = plt.subplots(2, 1, figsize=(10, 5), sharex=True) axes[0].plot(t, ir_ac, color="#3E6259", lw=1.2) axes[0].set_ylabel("IR AC") axes[0].grid(True, alpha=0.3) axes[0].set_title(f"{fname} ({label})" + (f" — {note}" if note else "")) axes[1].plot(t, red_ac, color="#D9663F", lw=1.2) axes[1].set_ylabel("Red AC") axes[1].set_xlabel("時間 (秒)") axes[1].grid(True, alpha=0.3) plt.tight_layout() out_path = os.path.join(out_dir, fname.replace(".npy", "_waveform.png")) plt.savefig(out_path, dpi=150) plt.close(fig) return out_path def plot_overview_grid(records, out_dir): """把所有檔案的處理後波型畫在同一張大圖裡，方便互相比較""" n = len(records) if n == 0: return None cols = 2 rows = int(np.ceil(n / cols)) fig, axes = plt.subplots(rows, cols, figsize=(12, 3 * rows), squeeze=False) for i, (fname, label, ir_ac, red_ac, note) in enumerate(records): r, c = divmod(i, cols) ax = axes[r][c] t = np.arange(len(ir_ac)) / FS ax.plot(t, ir_ac, color="#3E6259", lw=1.0, label="IR AC") ax.plot(t, red_ac, color="#D9663F", lw=1.0, label="Red AC") ax.set_title(f"{label}" + (f" ({note})" if note else ""), fontsize=9) ax.grid(True, alpha=0.3) ax.legend(fontsize=7) if r == rows - 1: ax.set_xlabel("秒") # 關掉多餘的空子圖 for i in range(n, rows * cols): r, c = divmod(i, cols) axes[r][c].axis("off") plt.tight_layout() out_path = os.path.join(out_dir, "waveform_overview.png") plt.savefig(out_path, dpi=150) plt.close(fig) return out_path # ============================================================ # 主流程 # ============================================================ files = sorted([f for f in os.listdir(BASE) if f.endswith(".npy")]) if not files: print(f"⚠️ 在 {BASE} 找不到任何 .npy 檔，請確認路徑/檔名") exit() print(f"共找到 {len(files)} 筆 .npy 檔\n") print(f"{'檔案':<38} {'A_ir':>10} {'A_red':>10} {'R1':>8} {'SpO2':>7} {'Hb':>8} 備註") print("-" * 100) all_labels = [] all_hbs = [] all_valid = [] waveform_records = [] # 給 overview 用: (fname, label, ir_ac, red_ac, note) waveform_dir = os.path.join(BASE, WAVEFORM_DIR_NAME) if PLOT_WAVEFORM: os.makedirs(waveform_dir, exist_ok=True) for fname in files: data = np.load(os.path.join(BASE, fname), allow_pickle=True).astype(float) ir_raw, red_raw = data[0], data[1] label = parse_time_label(fname) result, ir_ac, red_ac = get_hb_features(ir_raw, red_raw) note_str = "" if result is None: print(f"{fname:<38} {'--':>10} {'--':>10} {'--':>8} {'--':>7} {'--':>8} DC 太小，訊號無效") all_labels.append(label) all_hbs.append(None) all_valid.append(False) note_str = "DC 太小" else: A_ir, A_red, R1 = result hb, spo2, note = calc_hb(A_ir, R1) hb_str = f"{hb:.2f}" if hb is not None else "--" note_str = note if note else "" print(f"{fname:<38} {A_ir:>10.6f} {A_red:>10.6f} {R1:>8.4f} {spo2:>7.1f} {hb_str:>8} {note_str}") all_labels.append(label) all_hbs.append(hb) all_valid.append(hb is not None) # ⭐ 不論最後 Hb 有沒有算出來，都畫出「處理後、算 Hb 前」的波型，方便對照哪裡出問題 if PLOT_WAVEFORM: wpath = plot_processed_waveform(fname, label, ir_ac, red_ac, waveform_dir, note=note_str) waveform_records.append((fname, label, ir_ac, red_ac, note_str)) print("\n(A_ir / A_red / R1 這幾欄格式跟 logcat 的 'Hb: A_ir=...
